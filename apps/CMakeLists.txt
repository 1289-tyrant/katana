function (compileApp name) 
  if(ARGN)
    set(Sources ${ARGN})
  else()
    file(GLOB Sources *.cpp)
  endif()
  #file(GLOB Sources *.cpp)
  add_executable(${name} ${Sources})
endfunction (compileApp)

function(app name)
  compileApp(${name} ${ARGN})
  linkRuntime(${name})
  install(TARGETS ${name} DESTINATION bin)
endfunction(app)
